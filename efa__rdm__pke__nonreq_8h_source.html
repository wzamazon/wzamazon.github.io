53 {
54  struct efa_rdm_handshake_hdr *handshake_hdr;

57  handshake_hdr = (struct efa_rdm_handshake_hdr *)pke->wiredata;

72 static inline
73 uint64_t *efa_rdm_pke_get_handshake_opt_host_id_ptr(struct efa_rdm_pke *pke)
74 {
75  struct efa_rdm_base_hdr *base_hdr = efa_rdm_pke_get_base_hdr(pke);
76  struct efa_rdm_handshake_hdr *handshake_hdr;
77  struct efa_rdm_handshake_opt_host_id_hdr *handshake_opt_host_id_hdr;
78  size_t offset = 0;
79 
80  if (base_hdr->type != EFA_RDM_HANDSHAKE_PKT || !(base_hdr->flags & EFA_RDM_HANDSHAKE_HOST_ID_HDR))
81  return NULL;
82 
83  handshake_hdr = (struct efa_rdm_handshake_hdr *)pke->wiredata;
84  assert(handshake_hdr->type == EFA_RDM_HANDSHAKE_PKT);
85 
86  offset += sizeof(struct efa_rdm_handshake_hdr) +
87  (handshake_hdr->nextra_p3 - 3) * sizeof(uint64_t);
88 
89  assert(handshake_hdr->flags & EFA_RDM_HANDSHAKE_HOST_ID_HDR);
90 
91  if (handshake_hdr->flags & EFA_RDM_PKT_CONNID_HDR) {
92  /* HOST_ID_HDR is always immediately after CONNID_HDR(if present) */
93  offset += sizeof(struct efa_rdm_handshake_opt_connid_hdr);
94  }
95 
96  handshake_opt_host_id_hdr = (struct efa_rdm_handshake_opt_host_id_hdr *)(pke->wiredata + offset);
97  return &handshake_opt_host_id_hdr->host_id;
98 }
