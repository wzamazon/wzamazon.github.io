68  [EFA_RDM_MEDIUM_MSGRTM_PKT] = offsetof(struct efa_rdm_medium_rtm_base_hdr, seg_offset),

72  [EFA_RDM_RUNTREAD_MSGRTM_PKT] = offsetof(struct efa_rdm_runtread_rtm_base_hdr, seg_offset),
73  [EFA_RDM_RUNTREAD_TAGRTM_PKT] = offsetof(struct efa_rdm_runtread_rtm_base_hdr, seg_offset),
74  };
75 
76  pkt_type = efa_rdm_pke_get_base_hdr(pke)->type;
77  assert(efa_rdm_pkt_type_contains_data(pkt_type));
78 
79  if (efa_rdm_pkt_type_contains_seg_offset(pkt_type)) {
80  /* all such packet types has been listed in the array */
81  hdr_offset = offset_of_seg_offset_in_header[pkt_type];
