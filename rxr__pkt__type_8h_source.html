56static inline
57struct rxr_handshake_opt_connid_hdr *rxr_get_handshake_opt_connid_hdr(void *pkt)
58{
59 struct rxr_handshake_hdr *handshake_hdr;
60 size_t base_hdr_size;
61
62 handshake_hdr = (struct rxr_handshake_hdr *)pkt;
63 assert(handshake_hdr->type == RXR_HANDSHAKE_PKT);
64 assert(handshake_hdr->flags & RXR_PKT_CONNID_HDR);
65 base_hdr_size = sizeof(struct rxr_handshake_hdr) +
66 (handshake_hdr->nextra_p3 - 3) * sizeof(uint64_t);
67 return (struct rxr_handshake_opt_connid_hdr *)((char *)pkt + base_hdr_size);
68}

147/*
148 * RMA context packet, used to differentiate the normal RMA read, normal RMA
149 * write, and the RMA read in two-sided large message transfer
150 * Implementation of the function is in rxr_pkt_type_misc.c
151 */
152struct rxr_rma_context_pkt {
153 uint8_t type;
154 uint8_t version;
155 uint16_t flags;
156 /* end of rxr_base_hdr */
157 uint32_t context_type;
158 uint32_t tx_id; /* used by write context */
159 uint32_t read_id; /* used by read context */
160 size_t seg_size; /* used by read context */
161};

164 RXR_READ_CONTEXT = 1,
165 RXR_WRITE_CONTEXT,
166};

235/* General packet type helper functions */
236static inline
237int rxr_pkt_type_contains_rma_iov(int pkt_type)
238{
239 switch (pkt_type) {
240 case RXR_EAGER_RTW_PKT:
241 case RXR_DC_EAGER_RTW_PKT:
242 case RXR_LONGCTS_RTW_PKT:
243 case RXR_DC_LONGCTS_RTW_PKT:
244 case RXR_LONGREAD_RTW_PKT:
245 case RXR_SHORT_RTR_PKT:
246 case RXR_LONGCTS_RTR_PKT:
247 case RXR_WRITE_RTA_PKT:
248 case RXR_DC_WRITE_RTA_PKT:
249 case RXR_FETCH_RTA_PKT:
250 case RXR_COMPARE_RTA_PKT:
251 return 1;
252 break;
253 default:
254 return 0;
255 break;
256 }
257}
