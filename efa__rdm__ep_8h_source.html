331 /*
332  * @brief: check whether we should use p2p for this transaction
333  *
334  * @param[in] ep efa_rdm_ep
335  * @param[in] efa_mr memory registration struct
336  *
337  * @return: 0 if p2p should not be used, 1 if it should, and negative FI code
338  * if the transfer should fail.
339  */
340 static inline
341 int efa_rdm_ep_use_p2p(struct efa_rdm_ep *efa_rdm_ep, struct efa_mr *efa_mr)
342 {
343  if (!efa_mr)
344  return 0;
345 
346  /*
347  * always send from host buffers if we have a descriptor
348  */
349  if (efa_mr->peer.iface == FI_HMEM_SYSTEM)
350  return 1;
351 
352  if (efa_rdm_ep_domain(efa_rdm_ep)->hmem_info[efa_mr->peer.iface].p2p_supported_by_device)
353  return (efa_rdm_ep->hmem_p2p_opt != FI_HMEM_P2P_DISABLED);
354 
355  if (efa_rdm_ep->hmem_p2p_opt == FI_HMEM_P2P_REQUIRED) {
356  EFA_WARN(FI_LOG_EP_CTRL,
357  "Peer to peer support is currently required, but not available.\n");
358  return -FI_ENOSYS;
359  }
360 
361  return 0;
362 }
