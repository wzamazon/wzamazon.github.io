152/* Universal flags that can be applied on "rxr_base_hdr.flags".
153 *
154 * Universal flags start from the last bit and goes backwards.
155 * Because "rxr_base_hdr.flags" is a 16-bits integer, the
156 * last bit is the 15th bit.
157 * Other than universal flags, each packet type defines its
158 * own set of flags, which generally starts from the 0th bit
159 * in "rxr_base_hdr.flags".

171/*
172 * @breif header format of CTS packet (Packet Type ID 3)
173 *
174 * CTS is used in long-CTS sub-protocols for flow control.
175 *
176 * It is sent from receiver to sender, and contains number of bytes
177 * receiver is ready to receive.
178 *
179 * long-CTS is used not only by two-sided communication but also
180 * by emulated write and emulated read protocols.
181 *
182 * In emulated write, requester is sender, and responder is receiver.
183 *
184 * In emulated read, requester is receiver, and responder is sender.
185 */
186struct rxr_cts_hdr {
187 uint8_t type;
188 uint8_t version;
189 uint16_t flags;
190 /* end of rxr_base_hdr */
191 union {
192 uint32_t connid; /* sender connection ID, set when RXR_PKT_CONNID_HDR is on */
193 uint32_t padding; /* otherwise, a padding space to 8 bytes */
194 };
195 uint32_t send_id; /* ID of the send opertaion on sender side */
196 uint32_t recv_id; /* ID of the receive operatin on receive side */
197 uint64_t recv_length; /* number of bytes receiver is ready to receive */
198};
199
200#if defined(static_assert) && defined(__x86_64__)
201static_assert(sizeof(struct rxr_cts_hdr) == 24, "rxr_cts_hdr check");
202#endif

216/*
217 * @brief header format of DATA packet header (Packet Type ID 4)
218 *
219 * DATA is used in long-CTS sub-protocols.
220 *
221 * It is sent from sender to receiver, and contains a segment
222 * of application data.
223 *
224 * long-CTS is used not only by two-sided communication but also
225 * by emulated write and emulated read protocols.
226 *
227 * In emulated write, requester is sender, and responder is receiver.
228 *
229 * In emulated read, requester is receiver, and responder is sender.
230 */
231struct rxr_data_hdr {
232 uint8_t type;
233 uint8_t version;
234 uint16_t flags;
235 /* end of rxr_base_hdr */
236 uint32_t recv_id; /* ID of the receive operation on receiver */
237 uint64_t seg_length;
238 uint64_t seg_offset;
239 /* optional connid header, present when RXR_PKT_CONNID_HDR is on */
240 struct rxr_data_opt_connid_hdr connid_hdr[0];
241};
242
243#if defined(static_assert) && defined(__x86_64__)
244static_assert(sizeof(struct rxr_data_hdr) == 24, "rxr_data_hdr check");
245#endif

284/*
285 * @brief format of the EOR packet. (Packet Type ID 7)
286 *
287 * EOR packet is used in long-read sub-protocols, which is
288 * part of the extra request: RDMA read based data transfer.
289 *
290 * It is sent from receiver to sender, to notify
291 * the finish of data transfer.
292 *
293 * long-read is used not only by two-sided communication but also
294 * by emulated write.
295 *
296 * In emulated write, requester is sender, and responder is receiver.
297 */
298struct rxr_eor_hdr {
299 uint8_t type;
300 uint8_t version;
301 uint16_t flags;
302 /* end of rxr_base_hdr */
303 uint32_t send_id; /* ID of the send operation on sender */
304 uint32_t recv_id; /* ID of the receive operation on receiver */
305 union {
306 uint32_t connid; /* sender connection ID, optional, set whne RXR_PKT_CONNID_HDR is on */
307 uint32_t padding; /* otherwise, a padding space to 8 bytes boundary */
308 };
309};
310
311#if defined(static_assert) && defined(__x86_64__)
312static_assert(sizeof(struct rxr_eor_hdr) == 16, "rxr_eor_hdr check");

554/* @brief rxr_rta_hdr are shared by 4 types of RTA:

580 };
