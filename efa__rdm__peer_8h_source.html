102 static inline
103 bool efa_rdm_peer_support_rdma_read(struct efa_rdm_peer *peer)
104 {
105  /* RDMA READ is an extra feature defined in version 4 (the base version).
106  * Because it is an extra feature, an EP will assume the peer does not support
107  * it before a handshake packet was received.
108  */
109  return (peer->flags & EFA_RDM_PEER_HANDSHAKE_RECEIVED) &&
110  (peer->extra_info[0] & EFA_RDM_EXTRA_FEATURE_RDMA_READ);
111 }
