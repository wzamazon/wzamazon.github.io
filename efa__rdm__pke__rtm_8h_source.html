85 static inline
86 uint64_t efa_rdm_pke_get_rtm_tag(struct efa_rdm_pke *pke)
87 {
88  size_t offset;
89  uint64_t *tagptr;
90 
91  /*
92  * In consideration of performance, this function did not cast header
93  * into different header types to get tag, but assume tag is always
94  * the last member of header.
95  */
96  offset = efa_rdm_pke_get_req_base_hdr_size(pke) - sizeof(uint64_t);
97  tagptr = (uint64_t *)(pke->wiredata + offset);
98  return *tagptr;
99 }

115  /* tag is always the last member */
