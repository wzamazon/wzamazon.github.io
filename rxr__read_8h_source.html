41/*
42 * read can used in 2 scenarios:
43 *
44 * 1. application posted a read request.
45 *
46 * 2. read message protocol is being used, receiver is going
47 * to post a read requst.
48 *
49 * 3. a packet entry with data has been received, and the
50 * receiving buffer is on GPU memroy. A read request is
51 * being posted to copy data to receiving buffer.
52 *
53 * To distinguish them, we use a pointer as context.
54 *
55 * For 1, the tx_entry is used as context
56 * For 2, the rx_entry is used as context
57 * For 3, the pkt_entry is used as context
58 *
59 * We also store rxr_read_context_type in read_entry to specify
60 * context type.
61 */
62enum rxr_read_context_type {
63 RXR_READ_CONTEXT_TX_ENTRY,
64 RXR_READ_CONTEXT_RX_ENTRY,
65 RXR_READ_CONTEXT_PKT_ENTRY,
66};
67
68enum rxr_read_entry_state {
69 RXR_RDMA_ENTRY_FREE = 0,
70 RXR_RDMA_ENTRY_CREATED,
71 RXR_RDMA_ENTRY_PENDING,
72 RXR_RDMA_ENTRY_SUBMITTED,
73};
