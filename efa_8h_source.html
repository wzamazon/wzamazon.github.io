150static inline
151int efa_str_to_ep_addr(const char *node, const char *service, struct efa_ep_addr *addr)
152{
153 int ret;
154
155 if (!node)
156 return -FI_EINVAL;
157
158 memset(addr, 0, sizeof(*addr));
159
160 ret = inet_pton(AF_INET6, node, addr->raw);
161 if (ret != 1)
162 return -FI_EINVAL;
163 if (service)
164 addr->qpn = atoi(service);
165
166 return 0;
167}

388static inline
389bool efa_peer_support_rdma_read(struct rdm_peer *peer)
390{
391 /* RDMA READ is an extra feature defined in version 4 (the base version).
392 * Because it is an extra feature, an EP will assume the peer does not support
393 * it before a handshake packet was received.
394 */
395 return (peer->flags & RXR_PEER_HANDSHAKE_RECEIVED) &&
396 (peer->extra_info[0] & RXR_EXTRA_FEATURE_RDMA_READ);
397}

497/*
498 * @brief: check whether we should use p2p for this transaction
499 *
500 * @param[in] ep efa_ep
501 * @param[in] efa_mr memory registration struct
502 *
503 * @return: 0 if p2p should not be used, 1 if it should, and negative FI code
504 * if the transfer should fail.
505 */
506static inline int efa_ep_use_p2p(struct efa_ep *ep, struct efa_mr *efa_mr)
507{
508 if (!efa_mr)
509 return 0;
510
511 /*
512 * always send from host buffers if we have a descriptor
513 */
514 if (efa_mr->peer.iface == FI_HMEM_SYSTEM)
515 return 1;
516
517 if (ep->domain->hmem_info[efa_mr->peer.iface].p2p_supported_by_device)
518 return (ep->hmem_p2p_opt != FI_HMEM_P2P_DISABLED);
519
520 if (ep->hmem_p2p_opt == FI_HMEM_P2P_REQUIRED) {
521 EFA_WARN(FI_LOG_EP_CTRL,
522 "Peer to peer support is currently required, but not available.\n");
523 return -FI_ENOSYS;
524 }
525
526 return 0;
527}
