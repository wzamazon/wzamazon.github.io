71static inline
72uint32_t rxr_pkt_msg_id(struct rxr_pkt_entry *pkt_entry)
73{
74 struct rxr_rtm_base_hdr *rtm_hdr;
75
76 rtm_hdr = rxr_get_rtm_base_hdr(pkt_entry->pkt);
77 /* only msg and atomic request has msg_id */
78 assert(rtm_hdr->flags & (RXR_REQ_MSG | RXR_REQ_ATOMIC));
79 return rtm_hdr->msg_id;
80}

84static inline
85uint64_t rxr_pkt_rtm_tag(struct rxr_pkt_entry *pkt_entry)
86{
87 size_t offset;
88 uint64_t *tagptr;
89
90 /*
91 * In consideration of performance, this function did not cast header
92 * into different header types to get tag, but assume tag is always
93 * the last member of header.
94 */
95 offset = rxr_pkt_req_base_hdr_size(pkt_entry) - sizeof(uint64_t);
96 tagptr = (uint64_t *)((char *)pkt_entry->pkt + offset);
97 return *tagptr;
98}

100static inline
101void rxr_pkt_rtm_settag(struct rxr_pkt_entry *pkt_entry, uint64_t tag)
102{
103 size_t offset;
104 uint64_t *tagptr;
105
106 offset = rxr_pkt_req_base_hdr_size(pkt_entry) - sizeof(uint64_t);
107 /* tag is always the last member */
108 tagptr = (uint64_t *)((char *)pkt_entry->pkt + offset);
109 *tagptr = tag;
110}
