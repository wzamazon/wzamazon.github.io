172 /*
173  * @breif header format of CTS packet (Packet Type ID 3)
174  *
175  * CTS is used in long-CTS sub-protocols for flow control.
176  *
177  * It is sent from receiver to sender, and contains number of bytes
178  * receiver is ready to receive.
179  *
180  * long-CTS is used not only by two-sided communication but also
181  * by emulated write and emulated read protocols.
182  *
183  * In emulated write, requester is sender, and responder is receiver.
184  *
185  * In emulated read, requester is receiver, and responder is sender.
186  */
187 struct efa_rdm_cts_hdr {
188  uint8_t type;
189  uint8_t version;
190  uint16_t flags;
191  /* end of efa_rdm_base_hdr */
192  union {
193  uint32_t connid; /* sender connection ID, set when EFA_RDM_PKT_CONNID_HDR is on */
194  uint32_t padding; /* otherwise, a padding space to 8 bytes */
195  };
196  uint32_t send_id; /* ID of the send opertaion on sender side */
197  uint32_t recv_id; /* ID of the receive operatin on receive side */
198  uint64_t recv_length; /* number of bytes receiver is ready to receive */
199 };
