75static inline void rxr_poison_mem_region(void *ptr, size_t size)
76{
77 uint32_t rxr_poison_value = 0xdeadbeef;
78 for (int i = 0; i < size / sizeof(rxr_poison_value); i++)
79 memcpy((uint32_t *)ptr + i, &rxr_poison_value, sizeof(rxr_poison_value));
80}

82static inline void rxr_poison_pkt_entry(struct rxr_pkt_entry *pkt_entry, size_t wiredata_size)
83{
84 rxr_poison_mem_region(pkt_entry->wiredata, wiredata_size);
85 /*
86 * Don't poison pkt_entry->wiredata, which is the last element in rxr_pkt_entry
87 * pkt_entry->wiredata is released when the pkt_entry is released
88 */
89 rxr_poison_mem_region(pkt_entry, sizeof(struct rxr_pkt_entry) - sizeof(char *));
90}

526static inline void rxr_release_rx_entry(struct rxr_ep *ep,
527 struct rxr_op_entry *rx_entry)
528{
529 struct rxr_pkt_entry *pkt_entry;
530 struct dlist_entry *tmp;
531
532 if (rx_entry->peer)
533 dlist_remove(&rx_entry->peer_entry);
534
535 dlist_remove(&rx_entry->ep_entry);
536
537 if (!dlist_empty(&rx_entry->queued_pkts)) {
538 dlist_foreach_container_safe(&rx_entry->queued_pkts,
539 struct rxr_pkt_entry,
540 pkt_entry, entry, tmp) {
541 rxr_pkt_entry_release_tx(ep, pkt_entry);
542 }
543 dlist_remove(&rx_entry->queued_rnr_entry);
544 }
545
546 if (rx_entry->rxr_flags & RXR_OP_ENTRY_QUEUED_CTRL)
547 dlist_remove(&rx_entry->queued_ctrl_entry);
548
549#ifdef ENABLE_EFA_POISONING
550 rxr_poison_mem_region(rx_entry,
551 sizeof(struct rxr_op_entry));
552#endif
553 rx_entry->state = RXR_OP_FREE;
554 ofi_buf_free(rx_entry);
555}

576static inline int rxr_ep_use_zcpy_rx(struct rxr_ep *ep, struct fi_info *info)
577{
578 return !(ep->util_ep.caps & FI_DIRECTED_RECV) &&
579 !(ep->util_ep.caps & FI_TAGGED) &&
580 !(ep->util_ep.caps & FI_ATOMIC) &&
581 (ep->max_msg_size <= ep->mtu_size - ep->max_proto_hdr_size) &&
582 !rxr_need_sas_ordering(ep) &&
583 info->mode & FI_MSG_PREFIX &&
584 rxr_env.use_zcpy_rx;
585}

660 ssize_t prov_errno)
661{
662 struct fi_eq_err_entry err_entry;
663 int ret = -FI_ENOEQ;
664
665 FI_WARN(&rxr_prov, FI_LOG_EQ,
666 "Writing error to EQ: err: %s (%zd) prov_errno: %s (%zd)\n",
667 fi_strerror(err), err,
668 efa_strerror(prov_errno), prov_errno);
669 if (ep->eq) {
670 memset(&err_entry, 0, sizeof(err_entry));
671 err_entry.err = err;
672 err_entry.prov_errno = prov_errno;
673 ret = fi_eq_write(&ep->eq->eq_fid, FI_NOTIFY,
674 &err_entry, sizeof(err_entry),
675 UTIL_FLAG_ERROR);
676
677 if (ret == sizeof(err_entry))
678 return;
679 }
680
681 FI_WARN(&rxr_prov, FI_LOG_EQ, "Unable to write to EQ\n");
682 fprintf(stderr,
683 "Libfabric EFA provider has encounterd an internal error:\n\n"
684 "Libfabric error: (%zd) %s\n"
685 "EFA internal error: (%zd) %s\n\n"
686 "Your application will now abort().\n",
687 err, fi_strerror(err),
688 prov_errno, efa_strerror(prov_errno));
689 abort();
690}
