216static inline
217struct rxr_op_entry *rxr_op_entry_of_pkt_entry(struct rxr_pkt_entry *pkt_entry)
218{
219 enum rxr_x_entry_type x_entry_type;
220 /*
221 * pkt_entry->x_entry can be NULL when the packet is a HANDSHAKE packet
222 */
223 if (!pkt_entry->x_entry)
224 return NULL;
225
226 x_entry_type = RXR_GET_X_ENTRY_TYPE(pkt_entry);
227 return (x_entry_type == RXR_TX_ENTRY || x_entry_type == RXR_RX_ENTRY) ? pkt_entry->x_entry : NULL;
228}
